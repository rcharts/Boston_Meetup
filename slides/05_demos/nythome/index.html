Visualize Data

require(rCharts)

# initialize chart and set path to chart library
p2 <- rCharts$new()
p2$setLib('nyt_home')

# pass description, plot data and groups variable
p2$set(
  description = "This data comes from the managers dataset 
  included in the R package PerformanceAnalytics.",
  data = data_m,
  groups = "manager",
  height = 0,
  width = 0
)
