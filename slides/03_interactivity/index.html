Server Side Interactivity

Shiny

library(shiny)
runApp(list(
  ui = pageWithSidebar(
    headerPanel(""),
    sidebarPanel(
      selectInput("x", "", names(mtcars), "wt")  
    ),
    mainPanel(
      chartOutput("r1", "polycharts")
    )
  ),
  server = function(input, output){
    output$r1 <- renderChart2({
      rPlot(x = input$x, y = "mpg", data = mtcars, type = 'point')
    })
  }
))
